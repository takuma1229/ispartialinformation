def highlight_boolean(val):
    """True/Falseを色分けする"""
    color = 'lightgreen' if val is True else 'lightcoral'
    return f'background-color: {color}'

def style_dataframe(df):
    styled_df = df.style.applymap(highlight_boolean, subset=df.select_dtypes(include=[bool]).columns)
    styled_df = styled_df.set_caption("Experiment Data Visualization")

    # 縦書きスタイルを適用
    styled_df = styled_df.set_table_styles([
        {"selector": "th", "props": [
            ("writing-mode", "vertical-rl"),
            ("transform", "rotate(180deg)"),
            ("text-align", "center")
        ]}
    ])
    return styled_df
